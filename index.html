function jarvisMath(input) { // Girdiyi temizle ve küçük harfe çevir let text = input.toLowerCase().trim(); // 1. Kelime bazlı operatörleri sembole çevir const wordToSymbol = { "artı": "+", "topla": "+", "eksi": "-", "çıkar": "-", "çarpı": "*", "çarp": "*", "x": "*", // 'x' harfini çarpma yap "bölü": "/", "böl": "/", ":": "/" // ':' işaretini bölme yap }; for (let key in wordToSymbol) { // Kelimeleri (veya x gibi sembolleri) gerçek operatörlere dönüştür text = text.replace(new RegExp(key, 'g'), wordToSymbol[key]); } // 2. Matematiksel kalıbı bul: (Sayı) (Operatör) (Sayı) // Bu regex: "15+25", "10 x 5", "100/2" gibi her şeyi yakalar let match = text.match(/(\d+)\s*([\+\-\*\/])\s*(\d+)/); if (match) { try { const num1 = parseFloat(match[1]); const operator = match[2]; const num2 = parseFloat(match[3]); let result; // Güvenli hesaplama (eval yerine switch) switch (operator) { case '+': result = num1 + num2; break; case '-': result = num1 - num2; break; case '*': result = num1 * num2; break; case '/': result = num2 !== 0 ? num1 / num2 : "Sıfıra bölünemez"; break; } let response = `İşlem sonucu: ${result}, Efendim.`; // Arayüzü güncelle document.getElementById('transcript-text').innerText = response; // Sesli yanıt (Switch açıksa) if (document.getElementById('voice-switch').checked) { speak(response); } return true; } catch (e) { return false; } } return false; }
