// Matematiksel kelimeleri sembollere çeviren sözlük const mathMap = { "artı": "+", "topla": "+", "toplamı": "+", "eksi": "-", "çıkar": "-", "farkı": "-", "çarpı": "*", "çarp": "*", "çarpımı": "*", "bölü": "/", "böl": "/", "oranı": "/" }; function jarvisMath(input) { let text = input.toLowerCase(); // Kelimeleri sembollere çevir (artı -> +) for (let key in mathMap) { text = text.replace(new RegExp(key, 'g'), mathMap[key]); } // Sadece sayıları ve operatörleri ayıkla (Örn: "15 + 25 kaçtır" -> "15+25") let match = text.match(/\d+\s*[\+\-\*\/]\s*\d+/); if (match) { try { let result = eval(match[0]); // Basit işlemler için güvenlidir let response = `İşlem sonucu: ${result}, Efendim.`; // Ekranda göster document.getElementById('transcript-text').innerText = response; // Sesli yanıt ver (Eğer switch açıksa) if (document.getElementById('voice-switch').checked) { speak(response); } return true; } catch (e) { console.error("Matematik hatası:", e); return false; } } return false; } // Sesli yanıt fonksiyonu function speak(text) { const synth = window.speechSynthesis; const utterance = new SpeechSynthesisUtterance(text); utterance.lang = 'tr-TR'; // Cinsiyet seçimine göre ses ayarı buraya eklenebilir synth.speak(utterance); } // Klavye girişini yakalama (Mevcut handleChat fonksiyonunu güncelle) function handleChat(event) { if (event.key === "Enter") { let input = event.target.value; if (!jarvisMath(input)) { document.getElementById('transcript-text').innerText = "Bunu anlayamadım Efendim."; } event.target.value = ""; // Inputu temizle } }
