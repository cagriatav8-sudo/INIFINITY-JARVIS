Türkiye'deki tekstil sektörü yeniden doğal pamuğa yönelmeye başladı.", "Spor: Bugün Türkiye'de derbi heyecanı var! Ayrıca Türkiye'de TÜBİTAK bilim olimpiyatları başvuruları güncellendi.", "Gündem ve Doğa: Bugün sabah saatlerinde Bingöl Yedisu merkezli 4.4 büyüklüğünde bir deprem meydana geldi, çevre illerde de hissedildi.", "Hava Durumu: Meteoroloji bugün için yedi ilde gök gürültülü sağanak yağış ve Marmara bölgesi için fırtına uyarısı yaptı." ]; } function announceStatus(level) { let message = ""; if (level > 50) { message = "Selam! Bataryan gayet iyi durumda, enerji doluyuz."; } else if (level > 20 && level <= 50) { message = "Haberler orta şekerli, batarya yarıya indi haberin olsun."; } else if (level > 10 && level <= 20) { message = "Dikkat et knk, şarj azalıyor. Kırmızı bölgeye giriyoruz."; } else { message = "Eyvah! Pil bitmek üzere, hemen şarja tak yoksa kapanacağım."; } const statusElement = document.getElementById('status-text'); if (statusElement) { statusElement.innerText = message; } window.speechSynthesis.cancel(); const speech = new SpeechSynthesisUtterance(message); speech.lang = 'tr-TR'; speech.rate = 1; window.speechSynthesis.speak(speech); } function applyTheme(level) { const root = document.documentElement; if (level <= 10) { root.style.transition = "all 0.3s ease-out"; root.style.setProperty('--main', '#ffffff'); root.style.setProperty('--accent', '#ffffff'); return; } root.style.transition = "all 2s ease"; if (level > 50) { root.style.setProperty('--main', '#00aaff'); root.style.setProperty('--accent', '#00eeff'); } else if (level > 20 && level <= 50) { root.style.setProperty('--main', '#00ff88'); root.style.setProperty('--accent', '#aaff00'); } else if (level > 10 && level <= 20) { root.style.setProperty('--main', '#ff4444'); root.style.setProperty('--accent', '#ff8800'); } } function startListening() { const SpeechRecognition = window.SpeechRecognition || window.webkitSpeechRecognition; if (!SpeechRecognition) { alert("Tarayıcınız ses tanımayı desteklemiyor."); return; } const recognition = new SpeechRecognition(); recognition.lang = 'tr-TR'; recognition.interimResults = false; recognition.onstart = () => { const circle = document.getElementById('blue-circle'); if (circle) circle.style.boxShadow = "0 0 20px #00aaff"; }; recognition.onresult = (event) => { const text = event.results[0][0].transcript.toLowerCase(); if (text.includes("haber") || text.includes("neler oluyor")) { const news = getDailyNews(); let fullNews = "Efendim, bugünün öne çıkan haberleri şunlar: " + news.join(" "); const speech = new SpeechSynthesisUtterance(fullNews); speech.lang = 'tr-TR'; window.speechSynthesis.speak(speech); } else if (text.includes("batarya") || text.includes("pil")) { if (navigator.getBattery) { navigator.getBattery().then(b => { const level = Math.round(b.level * 100); announceStatus(level); }); } } else { const reply = new SpeechSynthesisUtterance("Sizi dinliyorum Efendim, haberleri duymak isterseniz haberler demeniz yeterli."); reply.lang = 'tr-TR'; window.speechSynthesis.speak(reply); } }; recognition.onend = () => { const circle = document.getElementById('blue-circle'); if (circle) circle.style.boxShadow = "none"; }; recognition.start(); } const blueCircle = document.getElementById('blue-circle'); if (blueCircle) { blueCircle.addEventListener('click', () => { updateBattery(); startListening(); }); } updateBattery();
